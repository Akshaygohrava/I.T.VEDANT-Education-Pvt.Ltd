// Program to check wheather prime or not

function isPrime(n) {
    let count = 0;
    for (let i = 1; i <= n; i++) {
       if (n % i == 0) {
         count++;
       }
        
    }
    return count == 2;
}
document.writeln("The Number 75 is prime or Not :" ,isPrime(75));

document.writeln("

");
